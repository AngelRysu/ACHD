sesiones

CREATE TABLE `sesiones` (
  `id` int NOT NULL AUTO_INCREMENT,
  `userID` int DEFAULT NULL,
  `sessionID` text,
  `active` tinyint(1) DEFAULT NULL,
  `expires` datetime(6) DEFAULT NULL,
  `created_at` datetime(6) DEFAULT NULL,
  `updated_at` datetime(6) DEFAULT NULL,
  PRIMARY KEY (`id`)
);


usersPrueba

CREATE TABLE `usersPrueba` (
  `id` int NOT NULL AUTO_INCREMENT,
  `user` text,
  `password` text,
  `nombres` text,
  `apellidos` text,
  `email` text,
  disponibilidad JSON,
  PRIMARY KEY (`id`)
);


CREATE TABLE asignaturas (clave VARCHAR(16) PRIMARY KEY, nombre TEXT, ht INT, hp INT, cr INT, grado INT, programa TEXT, sede TEXT, created_at DATETIME(6), updated_at DATETIME(6));

CREATE TABLE `docentes` (
  `id` int NOT NULL AUTO_INCREMENT,
  `nombre` text,
  `created_at` datetime(6) DEFAULT NULL,
  `updated_at` datetime(6) DEFAULT NULL,
  PRIMARY KEY (`id`)
);
